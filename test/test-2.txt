let a = [1,2,3,4,5,6], n = 6;

function sum(a, n) {
	let s = 0;
	for (let i = 0; i < n; i++) {
		s += a[i];
	}
	return s;
}

function selectSort(a, n) {
	for (let i = 0; i < n; i++) {
		let min_pos = a[i];
		for (let j = i + 1; j < n; j++)
			if (a[j] < a[min_pos])
				min_pos = j;
		let t = a[i];
		a[i] = a[min_pos];
		a[min_pos] = t;
	}
}
